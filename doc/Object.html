    # File knx/knx_tools.rb, line 28
def addr2str(a, ga=false)
    str=""
    if a.is_a?Array then
        #~ a = a[0]*256+a[1]
        a = a.pack('c*').unpack('n')[0]
    end
    if (ga)
        str = "#{(a >> 11) & 0xf}/#{(a >> 8) & 0x7}/#{a & 0xff}"
    else
        str = "#{a >> 12}.#{(a >> 8) & 0xf}.#{a & 0xff}"
    end
    return(str)
end

def decode_framedata(data)
    case data
    when Fixnum then "0x"+data.to_s(16).upcase
    when Array then data.collect{|b| "0x"+b.to_s(16).upcase}
    when String then data.unpack('C*').collect{|b| "0x"+b.to_s(16).upcase}
    else data
    end
end

def frame_inspect(frame)
    return "#{Time.now.strftime('%a %d/%m/%Y %H:%M:%S %Z')}: #{Ansible::KNX::APCICODES[frame.apci]}" +  
    " prio=#{Ansible::KNX::PRIOCLASSES[frame.prio_class]}" + " len=#{frame.datalength}" +
    " from #{addr2str(frame.src_addr)} to #{addr2str(frame.dst_addr, frame.daf)}" +
    " data=" + decode_framedata(frame.datalength > 1 ? frame.data : frame.apci_data).inspect   
end

def parse_ozw_headers(cpp_src)
    notificationtypes, valuegenres, valuetypes = [], [], []
    [   
        [File.join(cpp_src, "Notification.h"), "NotificationType", notificationtypes],
        [File.join(cpp_src, "value_classes", "ValueID.h"), "ValueGenre", valuegenres],
        [File.join(cpp_src, "value_classes", "ValueID.h"), "ValueType", valuetypes]
    ].each { | headerfile, enum_name, enum_array |
        puts "Parsing #{headerfile} for enum #{enum_name}..."
        #~ puts enum_re % enum_name
        foo = File.open(headerfile).read
        if enum = Regexp.new(ENUM_RE % enum_name,  Regexp::EXTENDED | Regexp::IGNORECASE | Regexp::MULTILINE).match(foo) then
            index = 0
            #~ puts enum[1].inspect
            #~ puts '-----------------'
            enum[1].split("\n").each { |line|
                if md = ENUM_RE_LINE.match(line) then
                    #puts md[1..-1].inspect
                    index =  (md[2] and md[3].length > 0) ? md[3].to_i : index+1
                    key, value = md[1], md[4]
                    enum_array[index] = [key, value]
                        #~ puts "#{enum_name}[#{index}] = [#{key}, #{value}]"
                end
            }
        end
    }
    return notificationtypes, valuegenres, valuetypes
end

    str2addr: Parse an address string into an unsigned 16-bit integer

    # File knx/knx_tools.rb, line 43
def str2addr(s)
    if m = s.match(/(\d*)\/(\d*)\/(\d*)/)
        a, b, c = m[1].to_i, m[2].to_i, m[3].to_i
        return ((a & 0x01f) << 11) | ((b & 0x07) << 8) | ((c & 0xff))
    end
    if m = s.match(/(\d*)\/(\d*)/)
        a,b = m[1].to_i, m[2].to_i
        return ((a & 0x01f) << 11) | ((b & 0x7FF))
    end
    if s.start_with?("0x")
        return (s.to_i & 0xffff)
    end
end
