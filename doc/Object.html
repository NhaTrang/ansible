addr2str: Convert an integer to an EIB address string, in the form “1/2/3“ or “1.2.3“ 

    # File knx/knx_tools.rb, line 28
28: def addr2str(a, ga=false)
29:     str=""
30:     if a.is_a?Array then
31:         #~ a = a[0]*256+a[1]
32:         a = a.pack('c*').unpack('n')[0]
33:     end
34:     if (ga)
35:         str = "#{(a >> 11) & 0xf}/#{(a >> 8) & 0x7}/#{a & 0xff}"
36:     else
37:         str = "#{a >> 12}.#{(a >> 8) & 0xf}.#{a & 0xff}"
38:     end
39:     return(str)
40: end
